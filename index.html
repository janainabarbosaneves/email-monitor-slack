Código do gatilho de 15 minutos (Google Apps Script):
// Cole no Google Apps Script e configure o gatilho function monitorarEmails() { var SLACK_WEBHOOK_URL = 'SEU_WEBHOOK_URL_AQUI'; var CANAL_SLACK = '#notificacoes-email'; var INTERVALO_MIN = 15; // Calcula o horário de corte (15 minutos atrás) var agora = new Date(); var corte = new Date(agora.getTime() - INTERVALO_MIN * 60 * 1000); var ano = corte.getFullYear(); var mes = String(corte.getMonth() + 1).padStart(2, '0'); var dia = String(corte.getDate()).padStart(2, '0'); var dataStr = ano + '/' + mes + '/' + dia; var threads = GmailApp.search( 'is:unread after:' + dataStr, 0, 20 ); var enviados = 0; threads.forEach(function(thread) { var mensagens = thread.getMessages(); mensagens.forEach(function(msg) { // Ignora mensagens anteriores ao corte if (msg.getDate() < corte) return; if (msg.isRead()) return; var assunto = msg.getSubject() || '(sem assunto)'; var remetente = msg.getFrom(); var data = Utilities.formatDate( msg.getDate(), Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm' ); var payload = { channel: CANAL_SLACK, username: 'Email Monitor', icon_emoji: ':email:', text: '*📧 Novo e-mail recebido*', attachments: [{ color: '#6366f1', fields: [ { title: 'Assunto', value: assunto, short: false }, { title: 'De', value: remetente, short: true }, { title: 'Data', value: data, short: true } ], footer: 'Email Monitor · Apps Script', ts: Math.floor(msg.getDate().getTime() / 1000) }] }; UrlFetchApp.fetch(SLACK_WEBHOOK_URL, { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload) }); msg.markRead(); enviados++; }); }); Logger.log('Emails enviados ao Slack: ' + enviados); } // ── Como configurar o gatilho de 15 minutos ── // 1. Abra script.google.com e cole este código // 2. Clique em "Acionadores" (ícone de relógio) no menu lateral // 3. Clique em "+ Adicionar acionador" // 4. Função: monitorarEmails // 5. Origem do evento: baseado em tempo // 6. Tipo: temporizador por minuto → a cada 15 minutos // 7. Clique em Salvar e autorize as permissões
